dump objects info from JavaScript

2013-05-08

Dump JavaScript object for PyV8 and NodeJS.

POC

function dump(obj){
  for(var item in obj){
    if(typeof item === "Object"){
      Print("Object's source is:" + item.toSource());
      dump(item);
    }else if(typeof obj[item] ==="Object"){
      dump(obj[item]);
    }else{
      Print(item.toString());
    }
  }
}
